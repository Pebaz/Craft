[
	print: []

	struct:
	[
		point
		x
		y
	]

	def:
	[
		[ show-point the-point ]
		print: [$the-point.x $the-point.y]
	]

	def:
	[
		[move-point the-point]
		set: [$the-point.x +: [$the-point.x 4]]
	]

	set: [a new: [$point 0 0]]
	move-point: [byval: [$a]]
	move-point: [byval: [$a]]
	show-point: [byval: [$a]]
]
